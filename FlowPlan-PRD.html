class SchedulingService { async calculateCriticalPath(projectId: string) { // 1. Get all data const tasks = await this.getTasks(projectId); const deps = await this.getDependencies(projectId); const holidays = await this.getHolidays(projectId); const teamMembers = await this.getTeamMembers(projectId); const timeOff = await this.getTimeOff(projectId); // 2. Forward Pass (with resource availability) for (const task of this.topologicalSort(tasks, deps)) { const assignee = teamMembers.find(m => m.userId === task.assigneeId); const unavailableDays = this.getUnavailableDays(task.assigneeId, timeOff, holidays); task.es = this.calculateEarlyStart(task, deps, unavailableDays); task.ef = this.addWorkingDays(task.es, task.duration, unavailableDays, assignee); } // 3.
